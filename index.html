# 🤖 F&B Intel Bot — Binance Trading Bot # Yêu cầu: pip install python-binance pandas ta import os from binance.client import Client import pandas as pd, ta from time import sleep client = Client(os.getenv('BINANCE_API_KEY'), os.getenv('BINANCE_SECRET')) SYMBOL = 'BTCUSDT' TRADE_AMOUNT = 50 def get_signal(symbol='BTCUSDT', interval='1h', limit=100): klines = client.get_klines(symbol=symbol, interval=interval, limit=limit) df = pd.DataFrame(klines, columns=['t','o','h','l','c','v','ct','qa','nt','tb','tq','i']) df['c'] = df['c'].astype(float) df['ma20'] = df['c'].rolling(20).mean() df['ma50'] = df['c'].rolling(50).mean() df['rsi'] = ta.momentum.RSIIndicator(df['c'], window=14).rsi() last = df.iloc[-1] if last['ma20'] > last['ma50'] and last['rsi'] < 65: return 'BUY' elif last['ma20'] < last['ma50'] and last['rsi'] > 35: return 'SELL' return 'HOLD' while True: signal = get_signal() price = float(client.get_symbol_ticker(symbol=SYMBOL)['price']) print(f"Signal: {signal} | BTC: ${price:,.0f}") if signal == 'BUY': qty = round(TRADE_AMOUNT / price, 5) # client.order_market_buy(symbol=SYMBOL, quantity=qty) print(f" → MOCK BUY {qty} BTC @ ${price:,.0f}") sleep(3600)
📋 Hướng dẫn: (1) Tạo Binance API key (chỉ spot). (2) VPS Ubuntu $5/tháng. (3) pip install python-binance pandas ta. (4) Bỏ comment lệnh order để chạy thật. Mock trước!
