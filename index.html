# 🤖 F&B Intel Bot — Binance Trading Bot # Yêu cầu: pip install python-binance pandas ta # Cài đặt: thêm API key Binance vào .env import os from binance.client import Client import pandas as pd import ta from time import sleep API_KEY = os.getenv('BINANCE_API_KEY') API_SECRET = os.getenv('BINANCE_SECRET') client = Client(API_KEY, API_SECRET) SYMBOL = 'BTCUSDT' TRADE_AMOUNT = 50 # USD mỗi lệnh RSI_LOW = 35 RSI_HIGH = 65 def get_signal(symbol='BTCUSDT', interval='1h', limit=100): klines = client.get_klines(symbol=symbol, interval=interval, limit=limit) df = pd.DataFrame(klines, columns=['t','o','h','l','c','v','ct','qa','nt','tb','tq','i']) df['c'] = df['c'].astype(float) df['ma20'] = df['c'].rolling(20).mean() df['ma50'] = df['c'].rolling(50).mean() df['rsi'] = ta.momentum.RSIIndicator(df['c'], window=14).rsi() last = df.iloc[-1] if last['ma20'] > last['ma50'] and last['rsi'] < RSI_HIGH: return 'BUY' elif last['ma20'] < last['ma50'] and last['rsi'] > RSI_LOW: return 'SELL' return 'HOLD' while True: signal = get_signal() price = float(client.get_symbol_ticker(symbol=SYMBOL)['price']) print(f"Signal: {signal} | BTC: ${price:,.0f}") if signal == 'BUY': qty = round(TRADE_AMOUNT / price, 5) # client.order_market_buy(symbol=SYMBOL, quantity=qty) print(f" → MOCK BUY {qty} BTC @ ${price:,.0f}") sleep(3600) # chạy mỗi 1 giờ
📋 Hướng dẫn: (1) Tạo tài khoản Binance → bật API key (chỉ spot trading). (2) Thuê VPS Ubuntu $5/tháng (Vultr, DigitalOcean). (3) pip install python-binance pandas ta. (4) Bỏ comment dòng order để chạy thật. Bắt đầu mock trước!
